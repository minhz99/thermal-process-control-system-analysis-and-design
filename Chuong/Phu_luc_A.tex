\section*{Mã nguồn MATLAB mô phỏng hệ thống điều khiển lưu lượng dầu}

Phụ lục này trình bày toàn bộ mã nguồn MATLAB được sử dụng để nhận dạng đối tượng điều khiển, tổng hợp bộ điều khiển PID và mô phỏng đáp ứng quá độ của hệ thống điều khiển lưu lượng dầu nhiên liệu.

\subsection*{A.1 Khai báo tham số và hàm truyền đối tượng}

\begin{lstlisting}[language=Matlab, caption={Khai báo tham số hệ thống và hàm truyền đối tượng SOPDT}, label={lst:khai_bao}]
%% ========================================================
%  HE THONG DIEU KHIEN LUU LUONG DAU NHIEN LIEU
%  Mo phong MATLAB - Nhan dang doi tuong va tong hop PID
%  ========================================================
clc; clear; close all;

%% BUOC 1: Khai bao tham so doi tuong (nhan dang tu thuc nghiem)
K  = 0.294;   % He so truyen (Gain)
T  = 3.04;    % Hang so quan tinh T1 = T2 (s)
L  = 1.84;    % Thoi gian tre thuan (Dead time, s)

%% BUOC 2: Xay dung ham truyen doi tuong SOPDT
%   O(s) = K * exp(-L*s) / (1 + T*s)^2
s = tf('s');
G_plant_no_delay = K / (T*s + 1)^2;

% Xap xi Pade bac 1 cho khau tre exp(-L*s)
[num_pade, den_pade] = pade(L, 1);
G_delay = tf(num_pade, den_pade);

G_plant = G_plant_no_delay * G_delay;
fprintf('Ham truyen doi tuong (xap xi Pade bac 1):\n');
G_plant
\end{lstlisting}

\subsection*{A.2 Khai báo tham số bộ điều khiển PID}

\begin{lstlisting}[language=Matlab, caption={Tham số bộ điều khiển PID bền vững tối ưu}, label={lst:pid}]
%% BUOC 3: Tham so bo dieu khien PID (tinh toan theo phuong phap R1)
%   Kp = 8.33,  Ti = 6.073 s,  Td = 1.52 s
Kp = 8.33;
Ti = 6.073;
Td = 1.52;

% Ham truyen PID ly tuong:
%   R(s) = Kp * (1 + 1/(Ti*s) + Td*s)
R_pid = Kp * (1 + 1/(Ti*s) + Td*s);
fprintf('Ham truyen bo dieu khien PID:\n');
R_pid

%% BUOC 4: Xay dung he kin
%   G_cl(s) = R(s)*G(s) / (1 + R(s)*G(s))
G_open  = R_pid * G_plant;          % Ham truyen ho
G_closed = feedback(G_open, 1);     % Ham truyen kin
fprintf('Ham truyen he kin:\n');
G_closed
\end{lstlisting}

\subsection*{A.3 Mô phỏng đáp ứng quá độ}

\begin{lstlisting}[language=Matlab, caption={Vẽ đáp ứng bậc thang và phân tích chất lượng}, label={lst:dap_ung}]
%% BUOC 5: Mo phong dap ung bac thang (Step Response)
t_sim = 0:0.01:60;   % Thoi gian mo phong: 0 -- 60 giay

figure('Name', 'Dap ung qua do he kin', 'NumberTitle', 'off', ...
       'Position', [100, 100, 800, 500]);

[y, t] = step(G_closed, t_sim);
plot(t, y, 'b-', 'LineWidth', 2);
hold on;
yline(1.0, 'k--', 'Setpoint', 'LineWidth', 1.2);
yline(1.05, 'r:', '+5%', 'LineWidth', 1);
yline(0.95, 'r:', '-5%', 'LineWidth', 1);
hold off;

grid on;
title('Dap ung bac thang cua he thong dieu khien (vong kin)', ...
      'FontSize', 13, 'FontWeight', 'bold');
xlabel('Thoi gian (s)', 'FontSize', 11);
ylabel('Luu luong chuan hoa y(t)', 'FontSize', 11);
legend('Dap ung y(t)', 'Setpoint', 'Bien +-5%', 'Location', 'southeast');

%% BUOC 6: Phan tich chi tieu chat luong
info = stepinfo(G_closed);
fprintf('\n===== Chi tieu chat luong he thong =====\n');
fprintf('Thoi gian dat dinh (Peak time)  Tp = %.2f s\n', info.PeakTime);
fprintf('Do vot lo (Overshoot)           Os = %.2f %%\n', info.Overshoot);
fprintf('Thoi gian qua do (Settling)     Ts = %.2f s\n', info.SettlingTime);
fprintf('Bien do dinh                        = %.4f\n', info.Peak);
fprintf('Sai lech tinh (ess)             ~  0 (co tich phan I)\n');
\end{lstlisting}

\subsection*{A.4 Phân tích bền vững -- Đồ thị Bode}

\begin{lstlisting}[language=Matlab, caption={Vẽ đồ thị Bode và xác định biên độ dự trữ, pha dự trữ}, label={lst:bode}]
%% BUOC 7: Ve do thi Bode cua ham truyen ho G_open
figure('Name', 'Do thi Bode - Ham truyen ho', 'NumberTitle', 'off', ...
       'Position', [100, 100, 800, 600]);

margin(G_open);
grid on;
title('Do thi Bode - Ham truyen ho R(s)*G(s)', ...
      'FontSize', 13, 'FontWeight', 'bold');

% In bien do du tru va pha du tru
[Gm, Pm, Wcg, Wcp] = margin(G_open);
fprintf('\n===== Do ben vung he thong =====\n');
fprintf('Bien do du tru  Gm = %.2f dB  (tai w = %.3f rad/s)\n', ...
        20*log10(Gm), Wcg);
fprintf('Pha du tru      Pm = %.2f deg (tai w = %.3f rad/s)\n', Pm, Wcp);
if 20*log10(Gm) > 6 && Pm > 30
    fprintf('=> He THOA MAN yeu cau ben vung (Gm>6dB, Pm>30 deg)\n');
else
    fprintf('=> CANH BAO: He co the khong du ben vung!\n');
end
\end{lstlisting}

\subsection*{A.5 Mô phỏng đáp ứng xung (Impulse Response)}

\end{lstlisting}
